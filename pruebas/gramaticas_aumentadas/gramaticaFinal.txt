CompilationUnit -> PackageDecl ImportDecls TypeDecls | ImportDecls TypeDecls

ImportDecl -> IMPORT QualifiedName PUNTO ASTERISK PUNTO_Y_COMA
TypeDecls -> TypeDecl TypeDecls | ε
TypeDecl -> ClassDecl | InterfaceDecl | EnumDecl | PUNTO_Y_COMA
Modifiers -> Modifier Modifiers | ε
Modifier -> PUBLIC | PRIVATE | PROTECTED | ABSTRACT | STATIC | FINAL | SYNCHRONIZED | NATIVE | TRANSIENT | VOLATILE
ClassDecl -> Modifiers CLASS IDENTIFICADOR TypeParamsOpt ExtendsOpt ImplementsOpt ClassBody
InterfaceDecl -> Modifiers INTERFACE IDENTIFICADOR TypeParamsOpt ExtendsOpt InterfaceBody
EnumDecl -> Modifiers ENUM IDENTIFICADOR EnumBody
TypeParamsOpt -> TypeParams | ε

TypeParamList -> TypeParam | TypeParam COMA TypeParamList

MemberDecl -> FieldDecl | MethodDecl | ClassDecl | InterfaceDecl | EnumDecl
FieldDecl -> Modifiers Type VariableDeclarators PUNTO_Y_COMA
VariableDeclarators -> VariableDeclarator | VariableDeclarator COMA VariableDeclarators
VariableDeclarator -> IDENTIFICADOR VarDimOpt VarInitOpt
VarDimOpt -> CORCHETE_IZQ CORCHETE_DER VarDimOpt | ε
VarInitOpt -> ASIGNACION VariableInitializer | ε
VariableInitializer -> Expression | ArrayInitializer
ArrayInitializer -> LLAVE_IZQ ArrayInitListOpt LLAVE_DER
ArrayInitListOpt -> ArrayInitList | ε
ArrayInitList -> VariableInitializer | VariableInitializer COMA ArrayInitList
MethodDecl -> Modifiers MethodHeader MethodBody
MethodHeader -> TypeOrVoid IDENTIFICADOR PARENTESIS_IZQ FormalParamListOpt PARENTESIS_DER ThrowsOpt

ThrowsOpt -> THROWS QualifiedNameList | ε
MethodBody -> Block | PUNTO_Y_COMA
ConstructorDecl -> Modifiers IDENTIFICADOR PARENTESIS_IZQ FormalParamListOpt PARENTESIS_DER ThrowsOpt ConstructorBody
ConstructorBody -> Block
FormalParamListOpt -> FormalParamList | ε
FormalParamList -> FormalParam | FormalParam COMA FormalParamList
FormalParam -> ParamModifiersOpt Type IDENTIFICADOR | ParamModifiersOpt Type ELLIPSIS IDENTIFICADOR
ParamModifiersOpt -> Modifier ParamModifiersOpt | ε
Block -> LLAVE_IZQ BlockStatementsOpt LLAVE_DER

BlockStatements -> BlockStatement BlockStatements | BlockStatement
BlockStatement -> LocalVarDeclStatement | Statement
LocalVarDeclStatement -> Modifiers Type VariableDeclarators PUNTO_Y_COMA
Statement -> Block | IF PARENTESIS_IZQ Expression PARENTESIS_DER Statement ELSE Statement | IF PARENTESIS_IZQ Expression PARENTESIS_DER Statement | SWITCH PARENTESIS_IZQ Expression PARENTESIS_DER Block | WHILE PARENTESIS_IZQ Expression PARENTESIS_DER Statement | DO Statement WHILE PARENTESIS_IZQ Expression PARENTESIS_DER PUNTO_Y_COMA | FOR PARENTESIS_IZQ ForInitOpt PUNTO_Y_COMA ForCondOpt PUNTO_Y_COMA ForUpdateOpt PARENTESIS_DER Statement | BREAK IDENTIFICADOR_OPT PUNTO_Y_COMA | CONTINUE IDENTIFICADOR_OPT PUNTO_Y_COMA | RETURN ExpressionOpt PUNTO_Y_COMA | THROW Expression PUNTO_Y_COMA | TRY Block CatchClausesOpt FinallyOpt | ExpressionStatement | PUNTO_Y_COMA
IDENTIFICADOR_OPT -> IDENTIFICADOR | ε
ExpressionOpt -> Expression | ε
CatchClausesOpt -> CatchClauses | ε
CatchClauses -> CatchClause CatchClauses | CatchClause

FinallyOpt -> FINALLY Block | ε
ForInitOpt -> ForInit | ε
ForInit -> LocalVarDecl | ExpressionList
LocalVarDecl -> Modifiers Type VariableDeclarators
ForCondOpt -> Expression | ε
ForUpdateOpt -> ExpressionList | ε
ExpressionList -> Expression | Expression COMA ExpressionList

ExpressionStatement -> StatementExpression PUNTO_Y_COMA

ConditionalOrExpression -> ConditionalAndExpression | ConditionalOrExpression OR ConditionalAndExpression
ConditionalAndExpression -> InclusiveOrExpression | ConditionalAndExpression AND InclusiveOrExpression
InclusiveOrExpression -> ExclusiveOrExpression | InclusiveOrExpression BITOR ExclusiveOrExpression
ExclusiveOrExpression -> AndExpression | ExclusiveOrExpression BITXOR AndExpression
AndExpression -> EqualityExpression | AndExpression BITAND EqualityExpression
EqualityExpression -> RelationalExpression | EqualityExpression IGUAL RelationalExpression | EqualityExpression DIFERENTE RelationalExpression

ArrayDimsOpt -> CORCHETE_IZQ CORCHETE_DER ArrayDimsOpt | ε
VariableDeclaratorInit -> ASIGNACION VariableInitializer | ε
